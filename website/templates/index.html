Handler
pub async fn index(
    State(ctx): State<AppContext>,
    pagination: PaginationParams,
) -> Result<impl IntoResponse> {
    let page = Select::from("posts")
        .columns(COLUMNS)
        .order_by("created_at", Desc)
        .fetch_all::<Post>(&ctx.db).await?;
    render!(PostIndex { page })
}
